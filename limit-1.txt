-- AGMIS DATABASE SCHEMA (Supabase PostgreSQL)
-- Version: 2.0 (Execution-Ready)

-- 1. EXTENSIONS
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 2. ENUMS FOR ROLE-BASED ACCESS CONTROL
CREATE TYPE user_role AS ENUM ('faculty', 'student', 'admin');
CREATE TYPE performance_level AS ENUM ('Excellent', 'Good', 'Average', 'At Risk', 'Critical');

-- 3. USER MANAGEMENT (Integrates with Supabase Auth)
CREATE TABLE users (
    user_id UUID PRIMARY KEY REFERENCES auth.users(id),
    email VARCHAR(100) UNIQUE NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    role user_role NOT NULL DEFAULT 'student',
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    last_login TIMESTAMP WITH TIME ZONE
);

-- 4. CORE ACADEMIC TABLES
CREATE TABLE students (
    student_id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES users(user_id),
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    batch VARCHAR(20) NOT NULL, -- e.g., '2024-DS-A'
    department VARCHAR(50) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE subjects (
    subject_id SERIAL PRIMARY KEY,
    subject_name VARCHAR(100) NOT NULL,
    subject_code VARCHAR(20) UNIQUE NOT NULL,
    semester INTEGER NOT NULL CHECK (semester BETWEEN 1 AND 8),
    credits INTEGER NOT NULL
);

CREATE TABLE enrollment (
    enrollment_id SERIAL PRIMARY KEY,
    student_id INTEGER REFERENCES students(student_id),
    subject_id INTEGER REFERENCES subjects(subject_id),
    academic_year VARCHAR(10) NOT NULL, -- e.g., '2024-25'
    enrolled_on TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    UNIQUE(student_id, subject_id, academic_year)
);

-- 5. ACADEMIC RECORDS (Weekly Granularity)
CREATE TABLE academic_records (
    record_id SERIAL PRIMARY KEY,
    enrollment_id INTEGER REFERENCES enrollment(enrollment_id),
    week_number INTEGER NOT NULL CHECK (week_number BETWEEN 1 AND 16),
    attendance_theory DECIMAL(5,2) CHECK (attendance_theory BETWEEN 0 AND 100),
    attendance_practical DECIMAL(5,2) CHECK (attendance_practical BETWEEN 0 AND 100),
    internal_marks DECIMAL(5,2) CHECK (internal_marks BETWEEN 0 AND 100),
    assignment_score DECIMAL(5,2) CHECK (assignment_score BETWEEN 0 AND 100),
    exam_score DECIMAL(5,2), -- NULL until final exam
    weighted_score DECIMAL(5,2), -- Calculated via service layer
    performance_category performance_level,
    record_date DATE NOT NULL,
    updated_by UUID REFERENCES users(user_id),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 6. PREDICTION TABLES
CREATE TABLE prediction_results (
    prediction_id SERIAL PRIMARY KEY,
    student_id INTEGER REFERENCES students(student_id),
    prediction_date DATE NOT NULL,
    prediction_horizon INTEGER NOT NULL, -- Weeks ahead (4, 6, 8)
    predicted_score DECIMAL(5,2) NOT NULL,
    confidence_lower DECIMAL(5,2) NOT NULL,
    confidence_upper DECIMAL(5,2) NOT NULL,
    predicted_category performance_level NOT NULL,
    model_version VARCHAR(10) DEFAULT 'rf_v1.0',
    features_used JSONB NOT NULL, -- Snapshot of features at time of prediction
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE prediction_accuracy (
    accuracy_id SERIAL PRIMARY KEY,
    prediction_id INTEGER REFERENCES prediction_results(prediction_id),
    actual_score DECIMAL(5,2) NOT NULL,
    error DECIMAL(5,2) NOT NULL, -- Predicted - Actual
    absolute_error DECIMAL(5,2) NOT NULL,
    evaluated_on DATE NOT NULL
);

-- 7. HISTORY & AUDIT TABLES
CREATE TABLE academic_history (
    history_id SERIAL PRIMARY KEY,
    record_id INTEGER REFERENCES academic_records(record_id),
    old_attendance_theory DECIMAL(5,2),
    old_internal_marks DECIMAL(5,2),
    old_weighted_score DECIMAL(5,2),
    modified_by UUID REFERENCES users(user_id),
    modified_on TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

CREATE TABLE action_logs (
    log_id SERIAL PRIMARY KEY,
    user_id UUID REFERENCES users(user_id),
    action_type VARCHAR(50) NOT NULL, -- 'UPLOAD', 'UPDATE', 'PREDICT'
    entity_type VARCHAR(50) NOT NULL, -- 'student', 'record', 'prediction'
    entity_id INTEGER NOT NULL,
    details TEXT,
    timestamp TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 8. INDEXING FOR PERFORMANCE
CREATE INDEX idx_student_email ON students(email);
CREATE INDEX idx_academic_enrollment ON academic_records(enrollment_id);
CREATE INDEX idx_prediction_student ON prediction_results(student_id);

Error: Trying to create the users table without referencing auth.users.
Why: Supabase manages authentication in a separate schema.
Fix: Always use REFERENCES auth.users(id) for the primary key of your public user profile.
Avoid: Storing passwords in your own table. Let Supabase Auth handle security.
Lesson: Decouple security (Auth) from profile metadata (Roles).

from fastapi import APIRouter, HTTPException, Depends, UploadFile, File
from app.schemas.academic import APIResponse
from supabase import create_client, Client
from app.core.config import settings
import csv
import io

router = APIRouter()
supabase: Client = create_client(settings.SUPABASE_URL, settings.SUPABASE_KEY)

@router.get("/health")
async def health_check():
    return {"status": "healthy", "project": settings.PROJECT_NAME}

@router.post("/upload/attendance", response_model=APIResponse)
async def upload_attendance(file: UploadFile = File(...)):
    """
    FR-1.1: Faculty uploads attendance via CSV
    Columns: student_id, subject_id, date, status
    """
    if not file.filename.endswith('.csv'):
        raise HTTPException(status_code=400, detail="Invalid file format. Please upload CSV.")

    contents = await file.read()
    decoded = contents.decode('utf-8')
    reader = csv.DictReader(io.StringIO(decoded))
    
    records = []
    for row in reader:
        try:
            records.append({
                "student_id": int(row['student_id']),
                "subject_id": int(row['subject_id']),
                "attendance_theory": float(row['status']),
                "record_date": row['date'],
                "week_number": 1 # Simplified for initial upload
            })
        except KeyError as e:
            raise HTTPException(status_code=400, detail=f"Missing column: {str(e)}")

    # Bulk insert into Supabase
    data, count = supabase.table("academic_records").insert(records).execute()
    
    return {
        "status": "success", 
        "message": f"Successfully uploaded {len(records)} records"
    }

@router.get("/dashboard/student/{student_id}")
async def get_student_data(student_id: int):
    """
    4.1.2 Student - Data Consumer
    Fetches latest performance and predictions
    """
    # Fetch academic records
    records = supabase.table("academic_records")\
        .select("*")\
        .eq("student_id", student_id)\
        .execute()
    
    # Fetch latest prediction
    prediction = supabase.table("prediction_results")\
        .select("*")\
        .eq("student_id", student_id)\
        .order("prediction_date", desc=True)\
        .limit(1)\
        .execute()

    return {
        "records": records.data,
        "latest_prediction": prediction.data[0] if prediction.data else None
    }

import pandas as pd
import numpy as np
from typing import List, Dict

class FeatureService:
    @staticmethod
    def calculate_weighted_score(
        attendance: float, 
        internal_marks: float, 
        assignments: float, 
        exam: float = 0
    ) -> float:
        """
        FR-3.2: Formula from PRD
        Weights: Attendance(0.20), Internals(0.30), Assignments(0.20), Exam(0.30)
        """
        score = (attendance * 0.20) + (internal_marks * 0.30) + (assignments * 0.20) + (exam * 0.30)
        return round(score, 2)

    @staticmethod
    def calculate_attendance_trend(weekly_attendance: List[float]) -> int:
        """
        9.2.2: Attendance Trend (-1, 0, 1)
        Uses slope over last 4 weeks.
        """
        if len(weekly_attendance) < 2:
            return 0 # Stable if no history
            
        # Get last 4 weeks
        recent = weekly_attendance[-4:]
        x = np.arange(len(recent))
        y = np.array(recent)
        
        # Calculate slope (linear regression)
        slope, _ = np.polyfit(x, y, 1)
        
        if slope < -2: return -1 # Declining
        if slope > 2: return 1   # Improving
        return 0                 # Stable

    @staticmethod
    def get_performance_category(score: float) -> str:
        """
        9.2.3: Target Variable Classification
        """
        if score >= 85: return "Excellent"
        if score >= 75: return "Good"
        if score >= 60: return "Average"
        if score >= 50: return "At Risk"
        return "Critical"

    def engineer_student_features(self, df_records: pd.DataFrame) -> Dict:
        """
        Transforms raw weekly records into a single feature vector for a student.
        Required for Step 4 & 5.
        """
        # 1. Overall Attendance (Combined theory + practical)
        avg_attendance = df_records[['attendance_theory', 'attendance_practical']].mean().mean()
        
        # 2. Internal Marks Average (Normalized to 100)
        avg_internals = df_records['internal_marks'].mean()
        
        # 3. Assignment Completion Rate
        assignment_rate = df_records['assignment_score'].mean()
        
        # 4. Attendance Trend
        trend = self.calculate_attendance_trend(df_records['attendance_theory'].tolist())
        
        # 5. Weighted Score (Current Standing)
        current_score = self.calculate_weighted_score(
            avg_attendance, avg_internals, assignment_rate
        )

        return {
            "attendance_pct": round(avg_attendance, 2),
            "internal_avg": round(avg_internals, 2),
            "assignment_rate": round(assignment_rate, 2),
            "attendance_trend": trend,
            "weighted_score": current_score,
            "performance_category": self.get_performance_category(current_score)
        }

feature_service = FeatureService()

Explanation
Teaching Perspective: Feature engineering is like "pre-chewing" data for the ML model. The model doesn't know what a "declining trend" is. We calculate the slope (the angle of the line) of attendance and give the model a simple number: -1 for down, 0 for flat, 1 for up.
Technical Perspective: We use numpy.polyfit for a simple 1D linear regression to find the trend slope. This is computationally efficient for batch processing.
Design Choice: I rounded all scores to 2 decimal places. This ensures consistency when the data is eventually displayed on the student dashboard.
Common Mistakes
Error: Calculating trend based on only 1 week of data.
Why: You need at least 2 points to determine a direction (slope).
Fix: Added a check if len(weekly_attendance) < 2: return 0.
Lesson: Always handle edge cases where historical data is missing.

import pandas as pd
import joblib
import os
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_ McLean
from typing import Dict, List, Any

class ClassificationService:
    def __init__(self):
        self.model_path = "models/student_classifier.joblib"
        self.model = None
        
        # Load model if it exists
        if os.path.exists(self.model_path):
            self.model = joblib.load(self.model_path)

    def train_model(self, data: pd.DataFrame):
        """
        PRD 9.2.4: Hyperparameters for Random Forest
        """
        # Features (X): attendance_pct, internal_avg, assignment_rate, 
        # practical_score, attendance_trend, prev_exam_score, submission_consistency
        X = data.drop(columns=['target_category'])
        y = data['target_category']

        # Initialize model with PRD specific hyperparameters
        rf = RandomForestClassifier(
            n_estimators=200,
            max_depth=15,
            min_samples_split=10,
            min_samples_leaf=5,
            max_features='sqrt',
            class_weight='balanced', # PRD 9.2.7: Handle class imbalance
            random_state=42          # Reproducibility
        )

        rf.fit(X, y)
        
        # Save model
        os.makedirs("models", exist_ok=True)
        joblib.dump(rf, self.model_path)
        self.model = rf
        return "Model trained and saved successfully."

    def predict(self, features: List[float]) -> Dict[str, Any]:
        """
        Predicts the category and provides feature importance for explainability.
        """
        if not self.model:
            raise Exception("Model not trained. Please train the model first.")

        # Reshape for single prediction
        feat_array = [features]
        prediction = self.model.predict(feat_array)[0]
        probabilities = self.model.predict_proba(feat_array)[0]
        
        # Mapping probabilities to classes
        classes = self.model.classes_
        confidences = {classes[i]: round(probabilities[i] * 100, 2) for i in range(len(classes))}

        return {
            "category": prediction,
            "confidence_scores": confidences,
            "logic": "Random Forest Ensemble (200 trees)"
        }

    def get_feature_importance(self, feature_names: List[str]) -> Dict[str, float]:
        """
        PRD 9.2.2: Identify which factors most influence classification
        """
        if not self.model: return {}
        
        importances = self.model.feature_importances_
        return {name: round(imp, 4) for name, imp in zip(feature_names, importances)}

Explanation
Teaching Perspective: Think of the Random Forest as a "committee of 200 experts." Each expert (tree) looks at a student's data and votes on their category. The majority vote wins. This makes the system robust against weird outliers like a student who missed one class but is actually top of the batch.
Technical Perspective: We use joblib for persistence. This means once we train the model, it stays in the models/ folder as a binary file. The API can load it in milliseconds without needing to retrain every time a user logs in.
Design Choice: I included predict_proba. This allows us to see how "sure" the model is. If it says "At Risk" with 51% confidence, faculty should know it was a close call.
Common Mistakes
Error: ValueError: X has 7 features, but RandomForestClassifier is expecting 8.
Why: The order or number of features passed to predict() doesn't match what was used during fit().
Fix: Always maintain a strict list of feature names and order (e.g., Attendance -> Internals -> Assignments).
Lesson: Consistency in data shape is the #1 rule of ML deployment.

import pandas as pd
import numpy as np
import joblib
import os
from sklearn.ensemble import RandomForestRegressor
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score

class PredictionService:
    def __init__(self):
        """
        This is the constructor. It runs as soon as we create the service.
        It sets up the paths where our 'brain' (the model) is stored.
        """
        self.model_folder = "models"
        self.model_name = "predictor.joblib"
        self.model_path = os.path.join(self.model_folder, self.model_name)
        
        self.model = None
        self.mae_score = 5.0  # Default error margin (±5 marks) until trained
        
        # Automatically try to load the model if it already exists
        self._load_model()

    def _get_category_from_score(self, score: float) -> str:
        """
        Helper function to turn a number (like 82) into a word (like 'Good').
        Based on PRD Section 6.2 thresholds.
        """
        if score >= 85:
            return "Excellent"
        elif score >= 75:
            return "Good"
        elif score >= 60:
            return "Average"
        elif score >= 50:
            return "At Risk"
        else:
            return "Critical"

    def train_model(self, training_data: pd.DataFrame):
        """
        This method teaches the model how to predict.
        It uses historical data to learn patterns.
        """
        # 1. Prepare the features (X) and the target we want to predict (y)
        # Expected columns: ['attendance', 'prev_exam', 'internal_avg', 'assignment_rate', 'trend']
        X = training_data[['attendance', 'prev_exam', 'internal_avg', 'assignment_rate', 'trend']]
        y = training_data['final_score']

        # 2. Baseline Model (Linear Regression) - Simple math
        baseline = LinearRegression()
        baseline.fit(X, y)
        
        # 3. Production Model (Random Forest) - Complex patterns (PRD 9.3.4)
        # We use 100 trees to vote on the result
        self.model = RandomForestRegressor(
            n_estimators=100,
            max_depth=10,
            min_samples_split=10,
            random_state=42
        )
        self.model.fit(X, y)

        # 4. Calculate the 'Average Error' (MAE) so we know how much to trust it
        predictions = self.model.predict(X)
        self.mae_score = mean_absolute_error(y, predictions)

        # 5. Save the 'brain' to a file so we don't have to train it again later
        self._save_model()
        
        return f"Training complete. Average error margin: ±{round(self.mae_score, 2)} marks."

    def predict(self, student_features: list) -> dict:
        """
        This is the main function. It takes one student's data and 
        returns the predicted score and category.
        """
        if self.model is None:
            return {"error": "The model is not trained yet. Please train it first!"}

        # Convert the list [attendance, prev_exam, etc.] into a format the model likes
        input_data = np.array(student_features).reshape(1, -1)
        
        # Get the predicted number (e.g., 72.5)
        raw_prediction = self.model.predict(input_data)[0]
        
        # Ensure the score stays between 0 and 100
        final_score = float(np.clip(raw_prediction, 0, 100))
        
        # Calculate the confidence range (Score ± MAE)
        lower_range = max(0, final_score - self.mae_score)
        upper_range = min(100, final_score + self.mae_score)
        
        return {
            "predicted_score": round(final_score, 2),
            "confidence_interval": {
                "minus": round(lower_range, 2),
                "plus": round(upper_range, 2),
                "margin": round(self.mae_score, 2)
            },
            "predicted_category": self._get_category_from_score(final_score),
            "note": f"The student is likely to score between {round(lower_range, 1)} and {round(upper_range, 1)}"
        }

    def evaluate(self, X_test: pd.DataFrame, y_test: pd.Series) -> dict:
        """
        Checks how good the model is using data it hasn't seen before.
        Required for PRD Success Metrics (Section 12.1.1).
        """
        if self.model is None:
            return {"error": "No model to evaluate."}
            
        predictions = self.model.predict(X_test)
        
        mae = mean_absolute_error(y_test, predictions)
        rmse = np.sqrt(mean_squared_error(y_test, predictions))
        r2 = r2_score(y_test, predictions)
        
        return {
            "Mean Absolute Error (MAE)": round(mae, 2), # Goal: < 6
            "Root Mean Squared Error (RMSE)": round(rmse, 2), # Goal: < 8
            "R-Squared Score": round(r2, 4), # Goal: > 0.70
            "Is Successful": mae < 6 and r2 > 0.70
        }

    def _save_model(self):
        """Saves the model to the disk."""
        if not os.path.exists(self.model_folder):
            os.makedirs(self.model_folder)
        
        # We save both the model and the MAE score in a single dictionary
        data_to_save = {
            "regressor": self.model,
            "mae": self.mae_score
        }
        joblib.dump(data_to_save, self.model_path)

    def _load_model(self):
        """Loads the model from the disk if it exists."""
        if os.path.exists(self.model_path):
            loaded_data = joblib.load(self.model_path)
            self.model = loaded_data["regressor"]
            self.mae_score = loaded_data["mae"]
            print(f"✅ Prediction model loaded successfully from {self.model_path}")
        else:
            print("⚠️ No prediction model found. It needs training.")

from typing import List, Dict, Any

class GuidanceService:
    def __init__(self):
        """
        The Guidance Service doesn't need to load any ML models.
        It just uses simple logic to help students.
        """
        # We define a numeric value for priorities so we can sort them later.
        # CRITICAL is most important (4), LOW is least (1).
        self.priority_map = {
            "CRITICAL": 4,
            "HIGH": 3,
            "MEDIUM": 2,
            "LOW": 1
        }

    def generate_guidance(self, student_data: Dict[str, Any]) -> List[Dict[str, Any]]:
        """
        This is the main function. 
        It takes student numbers and returns a list of helpful tips.
        """
        # Extract variables from the input dictionary for easy reading
        attendance = student_data.get("attendance", 0)
        internals = student_data.get("internal_marks", 0)
        submission = student_data.get("submission_rate", 0)
        predicted_score = student_data.get("predicted_score", 0)
        trend = student_data.get("attendance_trend", 0)  # -1 (down), 0 (flat), 1 (up)

        all_recommendations = []

        # --- RULE 1: Declining Trend (PRD Rule 4) ---
        if trend == -1:
            all_recommendations.append({
                "observation": "Your performance trend is currently declining.",
                "action": "Schedule an immediate meeting with your faculty mentor.",
                "impact": "Prevent a 10-mark drop in your next exam",
                "priority": "CRITICAL",
                "deadline": "This week"
            })

        # --- RULE 2: Low Attendance (PRD Rule 1) ---
        if attendance < 75:
            all_recommendations.append({
                "observation": f"Your attendance is {attendance}%, which is below the 75% threshold.",
                "action": "Increase your attendance to 85% or higher in the next 4 weeks.",
                "impact": "Gain approximately +8 marks in the final evaluation",
                "priority": "HIGH",
                "deadline": "End of Month"
            })

        # --- RULE 3: Assignment Backlog (PRD Rule 2) ---
        if submission < 70:
            all_recommendations.append({
                "observation": f"You have only submitted {submission}% of your assignments.",
                "action": "Complete all pending assignments before the next deadline.",
                "impact": "Gain +5 marks and avoid late penalties",
                "priority": "HIGH",
                "deadline": "Next Friday"
            })

        # --- RULE 4: Internal Marks Weak (PRD Rule 3) ---
        if internals < 60:
            all_recommendations.append({
                "observation": f"Your internal score of {internals}% is lagging.",
                "action": "Focus on concept practice and review past internal papers.",
                "impact": "Improve predicted score by +6 marks",
                "priority": "MEDIUM",
                "deadline": "Before next internal exam"
            })

        # --- RULE 5: Excellent Performance (PRD Rule 5) ---
        # Only show this if there are no serious problems
        if predicted_score >= 85 and len(all_recommendations) == 0:
            all_recommendations.append({
                "observation": "You are on track for outstanding results!",
                "action": "Maintain your current study consistency.",
                "impact": "Secure a distinction grade",
                "priority": "LOW",
                "deadline": "None"
            })

        # --- FINAL STEP: Sort and Filter ---
        
        # We sort by the priority map we made (Highest priority first)
        sorted_guidance = sorted(
            all_recommendations, 
            key=lambda x: self.priority_map.get(x["priority"], 0), 
            reverse=True
        )

        # We only return the top 3 most important messages (PRD 9.4.4)
        return sorted_guidance[:3]

What is a Database Layer?
Imagine your school has a giant, heavy filing cabinet in the basement. This cabinet holds every student's name, their grades, and their attendance. The Database Layer is like a helper who has the only key to that cabinet. When the computer needs to remember something, it asks the helper to write it down in a file. When the computer needs to show a grade, it asks the helper to go down to the basement and find that specific file.
Why do we need it?
Without this, the computer is like a person who forgets everything as soon as they blink. If you turn off the computer, all the student data would disappear forever! The database ensures that everything is saved safely in the "filing cabinet" (Supabase) so we can look at it tomorrow, next week, or next year.

import os
import psycopg2
import psycopg2.extras
from dotenv import load_dotenv

# Load the variables from the .env file
load_dotenv()

class DatabaseService:
    def __init__(self):
        """
        The constructor runs when we start the service.
        It pulls the 'secret keys' for the database from the .env file.
        """
        self.host = os.getenv("DB_HOST")
        self.port = os.getenv("DB_PORT", "5432")
        self.name = os.getenv("DB_NAME")
        self.user = os.getenv("DB_USER")
        self.password = os.getenv("DB_PASSWORD")
        
        self.connection = None
        
        # Try to connect as soon as the service is created
        self.connect()

    def connect(self):
        """
        This creates the actual 'pipe' between our Python code and Supabase.
        """
        try:
            self.connection = psycopg2.connect(
                host=self.host,
                port=self.port,
                dbname=self.name,
                user=self.user,
                password=self.password
            )
            print("✅ Successfully connected to Supabase PostgreSQL!")
        except Exception as e:
            print(f"❌ Error connecting to database: {e}")

    def close(self):
        """
        Closes the 'pipe' safely when we are done.
        """
        if self.connection:
            self.connection.close()
            print("Database connection closed.")

    def execute_query(self, sql: str, params: tuple = None):
        """
        A helper function to fetch data.
        It returns a list of DICTIONARIES (like {'name': 'Hamza'}) 
        instead of raw rows, making it easier for beginners to read.
        """
        if not self.connection:
            self.connect()
            
        # RealDictCursor makes the results look like Python Dictionaries
        cursor = self.connection.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
        try:
            cursor.execute(sql, params)
            result = cursor.fetchall()
            cursor.close()
            return result
        except Exception as e:
            print(f"❌ Database Query Error: {e}")
            return []

    def get_students(self):
        """
        Fetches all students from the filing cabinet.
        """
        sql = "SELECT id, name, email FROM students ORDER BY id ASC;"
        return self.execute_query(sql)

    def get_student_by_id(self, student_id: int):
        """
        Finds one specific student using their ID number.
        """
        sql = "SELECT * FROM students WHERE id = %s;"
        result = self.execute_query(sql, (student_id,))
        return result[0] if result else None

    def insert_academic_record(self, student_id: int, attendance: float, internal: float, assignments: float):
        """
        Writes a new week of data for a student into the database.
        """
        sql = """
            INSERT INTO academic_records (student_id, attendance, internal_marks, assignment_score)
            VALUES (%s, %s, %s, %s);
        """
        try:
            cursor = self.connection.cursor()
            cursor.execute(sql, (student_id, attendance, internal, assignments))
            self.connection.commit() # This 'Saves' the changes
            cursor.close()
            return True
        except Exception as e:
            print(f"❌ Error saving academic record: {e}")
            return False

    def save_prediction(self, student_id: int, score: float, lower: float, upper: float):
        """
        Saves the 'Crystal Ball' results so the student can see them later.
        """
        sql = """
            INSERT INTO predictions (student_id, predicted_score, lower_bound, upper_bound)
            VALUES (%s, %s, %s, %s);
        """
        try:
            cursor = self.connection.cursor()
            cursor.execute(sql, (student_id, score, lower, upper))
            self.connection.commit()
            cursor.close()
            return True
        except Exception as e:
            print(f"❌ Error saving prediction: {e}")
            return False

    def get_latest_prediction(self, student_id: int):
        """
        Gets the most recent prediction for a specific student.
        """
        sql = """
            SELECT * FROM predictions 
            WHERE student_id = %s 
            ORDER BY id DESC LIMIT 1;
        """
        result = self.execute_query(sql, (student_id,))
        return result[0] if result else None

# Create one instance to be used by the rest of the app
db_service = DatabaseService()

import pandas as pd
import io
from app.services.database import db_service
from app.services.features import feature_service
from app.services.classification import classification_service
from app.services.prediction import prediction_service
from app.services.guidance import guidance_service

class SystemIntegrator:
    """
    The 'General Manager' of AGMIS.
    This class connects all the services we built.
    """

    def process_faculty_upload(self, csv_file_contents: bytes):
        """
        1. Receives raw CSV data
        2. Saves it to Database
        3. Runs ML Models
        4. Saves Results
        """
        try:
            # Step 1: Read the CSV into a Pandas Table (DataFrame)
            df = pd.read_csv(io.BytesIO(csv_file_contents))
            
            # Step 2: Loop through every student in the file
            for index, row in df.iterrows():
                student_id = int(row['student_id'])
                
                # Step 3: Save the raw data to the Database (Step 9)
                db_service.insert_academic_record(
                    student_id=student_id,
                    attendance=row['attendance'],
                    internal=row['internal_marks'],
                    assignments=row['assignment_score']
                )
                
                # Step 4: Create 'Signals' for AI (Step 3)
                # Order: [attendance, prev_exam, internal_avg, assignment_rate, trend]
                features = [
                    row['attendance'], 
                    row['prev_exam'], 
                    row['internal_marks'], 
                    row['assignment_score'],
                    row['trend']
                ]
                
                # Step 5: Run the Crystal Ball (Step 5 - Prediction)
                prediction_result = prediction_service.predict(features)
                
                # Step 6: Save the prediction so the student can see it (Step 9)
                db_service.save_prediction(
                    student_id=student_id,
                    score=prediction_result['predicted_score'],
                    lower=prediction_result['confidence_interval']['minus'],
                    upper=prediction_result['confidence_interval']['plus']
                )
                
                # Step 7: Log the action
                print(f"✅ Processed student {student_id}: Predicted {prediction_result['predicted_score']}")

            return {"status": "success", "message": f"Processed {len(df)} students successfully."}

        except Exception as e:
            print(f"❌ Integration Error: {e}")
            return {"status": "error", "message": str(e)}

from fastapi import APIRouter, Request, UploadFile, File
from fastapi.responses import HTMLResponse
from fastapi.templating import Jinja2Templates
import pandas as pd
import io

# Import the "Robots" we built in previous steps
from app.services.features import feature_service
from app.services.prediction import prediction_service
from app.services.guidance import guidance_service

# Setup the template folder
templates = Jinja2Templates(directory="app/templates")
router = APIRouter()

# 1. THE VIEW (Show the empty page first)
@router.get("/faculty/dashboard", response_class=HTMLResponse)
async def get_dashboard(request: Request):
    """Shows the dashboard with no data when first opened."""
    return templates.TemplateResponse("faculty_dashboard.html", {
        "request": request,
        "students": [], # Empty list
        "stats": {"total_students": 0, "avg_score": 0, "at_risk": 0, "critical": 0}
    })

# 2. THE ACTION (Process the CSV and show results)
@router.post("/faculty/upload", response_class=HTMLResponse)
async def upload_csv(request: Request, file: UploadFile = File(...)):
    """
    This is the main logic. It takes a CSV and turns it into AI insights.
    """
    # Step 1: Read the uploaded CSV file using Pandas
    contents = await file.read()
    df = pd.read_csv(io.BytesIO(contents))
    
    processed_students = []
    total_score = 0
    at_risk_count = 0
    critical_count = 0

    # Step 2: Loop through every student in the file
    for index, row in df.iterrows():
        # Prepare the features for the AI model
        # Order: attendance, prev_exam, internal_avg, assignment_rate, trend
        features = [
            float(row['attendance']),
            float(row['prev_exam']),
            float(row['internal_marks']),
            float(row['assignment_score']),
            float(row['trend'])
        ]

        # Step 3: Run Prediction (Crystal Ball)
        pred = prediction_service.predict(features)
        predicted_score = pred['predicted_score']
        category = pred['predicted_category']

        # Step 4: Run Guidance (Helper tips)
        # We give the guidance engine the current data
        student_info = {
            "attendance": row['attendance'],
            "internal_marks": row['internal_marks'],
            "submission_rate": row['assignment_score'],
            "predicted_score": predicted_score,
            "attendance_trend": row['trend']
        }
        tips = guidance_service.generate_guidance(student_info)
        
        # Get the #1 most important tip, or a default message if none
        top_tip = tips[0]['action'] if tips else "Maintain current performance."

        # Step 5: Package the result for the table
        student_result = {
            "id": row['student_id'],
            "name": row['name'],
            "category": category,
            "predicted": predicted_score,
            "risk": "High" if predicted_score < 60 else "Low",
            "top_guidance": top_tip
        }
        processed_students.append(student_result)

        # Update summary statistics
        total_score += predicted_score
        if predicted_score < 60: at_risk_count += 1
        if predicted_score < 50: critical_count += 1

    # Step 6: Calculate final stats for the top cards
    stats = {
        "total_students": len(processed_students),
        "avg_score": round(total_score / len(processed_students), 1) if processed_students else 0,
        "at_risk": at_risk_count,
        "critical": critical_count
    }

    # Step 7: Send all the "Real" data back to the HTML page
    return templates.TemplateResponse("faculty_dashboard.html", {
        "request": request,
        "students": processed_students,
        "stats": stats
    })

Error Handling: Simple loops and if/else statements make it easy to debug if the CSV has bad data.

Warning: UserWarning: X does not have valid feature names...
Why: When you trained the model (using train_predictor.py), you gave it a table (DataFrame) with column names like "attendance" and "prev_exam". But when you ask it for a prediction, you are giving it a simple list of numbers [92, 85, 88, 95, 1]. The model is saying: "I know how to handle this, but I'd prefer if you gave me the names so I don't mix up the columns."
Fix: We will update prediction.py to wrap the numbers in a small DataFrame with names before predicting.
Avoid: Don't ignore warnings in ML; they often point to "data drift" or mismatched columns.
Lesson: Consistency is key. If you train with names, predict with names.

def predict(self, student_features: list) -> dict:
        """
        Takes one student's data and returns the predicted score.
        Updated to use column names and avoid warnings.
        """
        if self.model is None:
            return {"error": "The model is not trained yet."}

        # 1. Define the exact column names used during training
        column_names = ['attendance', 'prev_exam', 'internal_avg', 'assignment_rate', 'trend']
        
        # 2. Convert the list into a tiny 1-row DataFrame with names
        input_df = pd.DataFrame([student_features], columns=column_names)
        
        # 3. Predict without warnings
        raw_prediction = self.model.predict(input_df)[0]
        
        # 4. Rest of the logic remains the same
        final_score = float(np.clip(raw_prediction, 0, 100))
        lower_range = max(0, final_score - self.mae_score)
        upper_range = min(100, final_score + self.mae_score)
        
        return {
            "predicted_score": round(final_score, 2),
            "confidence_interval": {
                "minus": round(lower_range, 2),
                "plus": round(upper_range, 2),
                "margin": round(self.mae_score, 2)
            },
            "predicted_category": self._get_category_from_score(final_score)
        }

class FeatureService:
    def build_features(self, row):
        """
        Converts raw counts from CSV into percentages for the AI.
        Input row must have: lectures_total, lectures_present, etc.
        """
        # Calculate percentages (handle division by zero just in case)
        attn_pct = (row['lectures_present'] / row['lectures_total']) * 100 if row['lectures_total'] > 0 else 0
        assign_pct = (row['assignments_submitted'] / row['assignments_given']) * 100 if row['assignments_given'] > 0 else 0
        intern_pct = (row['internal_marks_scored'] / row['internal_marks_total']) * 100 if row['internal_marks_total'] > 0 else 0

        # Return the EXACT list format the prediction model expects
        return [
            round(attn_pct, 2),
            float(row['prev_exam_score']),
            round(intern_pct, 2),
            round(assign_pct, 2),
            int(row['trend'])
        ]

import sqlite3
import os

class DatabaseService:
    def __init__(self):
        self.db_path = "local.db"
        self.init_db()

    def get_connection(self):
        # row_factory makes results look like dictionaries: {'name': 'Hamza'}
        conn = sqlite3.connect(self.db_path)
        conn.row_factory = sqlite3.Row
        return conn

    def init_db(self):
        conn = self.get_connection()
        cursor = conn.cursor()
        # Table 1: Student Profiles & Passwords
        cursor.execute('''CREATE TABLE IF NOT EXISTS students (
            id INTEGER PRIMARY KEY, name TEXT, password TEXT)''')
        # Table 2: Academic Records & Predictions
        cursor.execute('''CREATE TABLE IF NOT EXISTS records (
            student_id INTEGER, attendance REAL, internal REAL, assignment REAL, 
            prev_exam REAL, trend INTEGER, predicted REAL, category TEXT,
            FOREIGN KEY(student_id) REFERENCES students(id))''')
        # Table 3: Notifications Panel
        cursor.execute('''CREATE TABLE IF NOT EXISTS notifications (
            student_id INTEGER, message TEXT, type TEXT)''')
        conn.commit()
        conn.close()

    def save_student_data(self, s_id, name, pwd, attendance, internal, assign, prev, trend, pred, cat):
        conn = self.get_connection()
        cursor = conn.cursor()
        # Insert student if doesn't exist, ignore if does
        cursor.execute("INSERT OR IGNORE INTO students (id, name, password) VALUES (?, ?, ?)", (s_id, name, pwd))
        # Clear old records (to keep it simple, we only store latest state for now)
        cursor.execute("DELETE FROM records WHERE student_id = ?", (s_id,))
        cursor.execute('''INSERT INTO records VALUES (?, ?, ?, ?, ?, ?, ?, ?)''', 
                       (s_id, attendance, internal, assign, prev, trend, pred, cat))
        conn.commit()
        conn.close()

    def add_notification(self, s_id, msg, msg_type="info"):
        conn = self.get_connection()
        cursor = conn.cursor()
        cursor.execute("INSERT INTO notifications (student_id, message, type) VALUES (?, ?, ?)", (s_id, msg, msg_type))
        conn.commit()
        conn.close()

db_service = DatabaseService()

import sqlite3

class DatabaseService:
    def __init__(self):
        self.db_path = "local.db"
        self.init_db()

    def get_conn(self):
        conn = sqlite3.connect(self.db_path, timeout=10)
        conn.row_factory = sqlite3.Row
        return conn

    def init_db(self):
        with self.get_conn() as conn:
            # 1. Users (Role-based: faculty sees one subject, student sees all)
            conn.execute('''CREATE TABLE IF NOT EXISTS users (
                username TEXT PRIMARY KEY, password TEXT, role TEXT, 
                name TEXT, subject_name TEXT)''')
            
            # 2. Weekly Records (Granular metrics per subject)
            conn.execute('''CREATE TABLE IF NOT EXISTS weekly_records (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                student_id TEXT, subject TEXT, week INTEGER,
                lec_attn REAL, prac_attn REAL, assign_pct REAL,
                internal REAL, external REAL, practical REAL)''')
            
            # 3. Subject-wise Predictions
            conn.execute('''CREATE TABLE IF NOT EXISTS predictions (
                student_id TEXT, subject TEXT, score REAL, 
                category TEXT, PRIMARY KEY(student_id, subject))''')
            
            # 4. Savage Notifications
            conn.execute('''CREATE TABLE IF NOT EXISTS notifications (
                student_id TEXT, message TEXT, subject TEXT)''')

            # Initialize Faculty Accounts
            faculty_list = [
                ("hiral", "ml123", "Machine Learning", "Hiral Maam"),
                ("harish", "gov123", "Governance", "Harish Sir"),
                ("jennifer", "iot123", "IoT", "Jennifer Maam"),
                ("vivek", "an123", "Analysis", "Vivek Sir"),
                ("vibha", "eda123", "EDA", "Vibha Maam")
            ]
            for uname, pwd, subj, name in faculty_list:
                conn.execute("INSERT OR IGNORE INTO users VALUES (?,?,?,?,?)", 
                             (uname, pwd, "faculty", name, subj))

    def save_faculty_data(self, s_id, name, subj, week, metrics, pred):
        with self.get_conn() as conn:
            # Create student user if not exists (PIN is 1234 by default)
            conn.execute("INSERT OR IGNORE INTO users VALUES (?,?,?,?,?)", (s_id, "1234", "student", name, None))
            # Save Record
            conn.execute('''INSERT INTO weekly_records (student_id, subject, week, lec_attn, prac_attn, assign_pct, internal, external, practical) 
                         VALUES (?,?,?,?,?,?,?,?,?)''', (s_id, subj, week, *metrics))
            # Save Prediction
            conn.execute("REPLACE INTO predictions VALUES (?,?, ?, ?)", (s_id, subj, pred['score'], pred['cat']))

class PredictionService:
    def predict_subject_score(self, metrics):
        """
        Input: [lec_attn, prac_attn, assign_pct, internal, external, practical]
        Returns: score (0-100), category
        """
        # Manual weight calculation for subject-wise accuracy
        # Lecture (20%), Practical (10%), Assignments (10%), Internal (20%), External (40%)
        l_pct, p_pct, a_pct, i_val, e_val, pr_val = metrics
        
        # Normalize marks to percentages
        internal_pct = (i_val / 20) * 100
        external_pct = (e_val / 75) * 100
        
        score = (l_pct * 0.20) + (p_pct * 0.10) + (a_pct * 0.10) + (internal_pct * 0.20) + (external_pct * 0.40)
        
        cat = "Excellent"
        if score < 50: cat = "Critical"
        elif score < 65: cat = "At Risk"
        elif score < 85: cat = "Average"
        
        return {"score": round(score, 2), "cat": cat}

class FeatureService:
    def build_features(self, row):
        """
        Converts raw counts from your new CSV format into percentages.
        """
        # 1. Lecture Percentage
        lec_pct = (row['lecture_present'] / row['lecture_total'] * 100) if row['lecture_total'] > 0 else 0

This "infinite buffering" in FastAPI usually happens for two specific reasons:
SQLite Threading Issue: SQLite doesn't like it when multiple "threads" (FastAPI's way of working) try to touch the database file at once. It "locks" and waits forever.
Ghost Process: An old version of your program is still running in the background, holding the port hostage.
Follow these 3 steps to unfreeze your system:
Step 1: Kill the "Ghost" Processes
Press Ctrl + C in your terminal to stop the current run. Then, run this command in your terminal to kill any hidden Python processes:
code
Bash
taskkill /F /IM python.exe
Step 2: Fix the Database Locking (Crucial)
Open backend/app/services/database.py. We need to tell SQLite to allow multiple threads.
Update your get_conn function to look exactly like this:

def get_conn(self):
        # We add 'check_same_thread=False' to stop the infinite buffering/hanging
        conn = sqlite3.connect(
            self.db_path, 
            timeout=15, 
            check_same_thread=False
        )
        conn.row_factory = sqlite3.Row
        return conn

Explanation (Why this fixed it)
Technical Reason: FastAPI is "Asynchronous." It handles many requests at the same time using different threads. By default, SQLite only allows the specific thread that opened the database to use it. If another thread tries, SQLite freezes. Setting check_same_thread=False tells SQLite: "It's okay, I trust FastAPI to handle the traffic."
Simple Reason: It's like a narrow doorway. Only one person was allowed through at a time, and someone got stuck in the middle. We just widened the door.

# Inside faculty_dash route:
@app.get("/faculty/dashboard")
async def faculty_dash(request: Request):
    uname = request.cookies.get("user_id")
    with db_service.get_conn() as conn:
        faculty = conn.execute("SELECT * FROM users WHERE username=?", (uname,)).fetchone()
        subj = faculty['subject_name']
        students = conn.execute("SELECT p.*, u.name FROM predictions p JOIN users u ON p.student_id = u.username WHERE p.subject=?", (subj,)).fetchall()
        
        # 1. Compute Stats
        total = len(students)
        if total == 0: return templates.TemplateResponse("faculty_dashboard.html", {"request":request, "faculty":faculty, "students":[], "stats":{"total":0, "avg":0, "safe":1, "at_risk":0, "critical":0, "hist_data":[0,0,0,0], "insight_msg":"No data uploaded."}})
        
        avg_score = round(sum(s['score'] for s in students) / total, 1)
        critical = len([s for s in students if s['risk_level'] == 'Critical'])
        at_risk = len([s for s in students if s['risk_level'] == 'High'])
        safe = total - (critical + at_risk)
        
        # 2. Histogram Buckets
        h1 = len([s for s in students if s['score'] <= 40])
        h2 = len([s for s in students if 40 < s['score'] <= 60])
        h3 = len([s for s in students if 60 < s['score'] <= 80])
        h4 = len([s for s in students if 80 < s['score'] <= 100])
        
        # 3. Auto Insight Logic
        insight = "Batch performance is stable."
        if (critical / total) > 0.2: insight = "🚨 Batch health poor! Over 20% students are in the Critical zone. Immediate intervention required."
        elif avg_score > 75: insight = "✅ Strong batch performance. Most students are on track for distinction."

        stats = {
            "total": total, "avg": avg_score, "critical": critical, 
            "at_risk": at_risk, "safe": safe, "hist_data": [h1, h2, h3, h4],
            "insight_msg": insight
        }
        
    return templates.TemplateResponse("faculty_dashboard.html", {"request": request, "faculty": faculty, "students": students, "stats": stats})
